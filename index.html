import numpy as np
import math

class Perceptron:
    def __init__(self, input_size, activation_function='step', learning_rate=0.1):
        self.weights = np.random.randn(input_size)
        self.bias = np.random.uniform(-1, 1)
        self.activation_function = activation_function
        self.learning_rate = learning_rate
        
    def activation(self, x):
        if self.activation_function == 'linear':
            return x
        elif self.activation_function == 'step':
            return 1 if x >= 0 else 0
        elif self.activation_function == 'sigmoid':
            x = np.clip(x, -500, 500)
            return 1 / (1 + math.exp(-x))
        elif self.activation_function == 'relu':
            return max(0, x)
        elif self.activation_function == 'tanh':
            return math.tanh(x)
        elif self.activation_function == 'softmax':
            exp_x = math.exp(x)
            return exp_x / (exp_x + 1)
    
    def predict(self, inputs):
        weighted_sum = np.dot(inputs, self.weights) + self.bias
        return self.activation(weighted_sum)
    
    def train(self, training_inputs, labels, epochs=100):
        for epoch in range(epochs):
            total_error = 0
            for inputs, label in zip(training_inputs, labels):
                prediction = self.predict(inputs)
                error = label - prediction
                total_error += abs(error)
                
                self.weights += self.learning_rate * error * np.array(inputs)
                self.bias += self.learning_rate * error
            
            if epoch % 10 == 0:
                print(f"√âpoca {epoch}, Error: {total_error:.4f}")
            
            if total_error == 0:
                break
